EasyChat 主要由python开发而成的即时通讯软件，支持局域网p2p连接，以及广域网的聊天室连接。

使用说明

进入界面后，可以看到标题，四个选项以及一个蓝色的选中标志。按w和s键可以改变选中选项，按enter进入。

局域网通信

进入局域网通信后，程序会自动搜索局域网下的ip,搜索过程需要一定的时间。

完成初始化后，进入选择界面，通过键入选项，进行选择：

文件夹下有一个ip_info.txt内存有局域网下的所有在线的ip(指能够ping通，并不代表对方已经启动该软件).

您可以选择尝试连接对方的IP或者作为服务器等待对方与你进行连接。

输入文字，回车即可将内容发送给对方，此时对方发送的信息会从缓存中输出，如果您只想看对方发送的内容，可以直接按回车刷新。

退出当前状态，可以输入:exit。

连接服务器

连接服务器可以使局域网外的用户能够发送和接收消息。

接着需要输入一个昵称。

同样的，结束状态可以输入:exit,对方则会收到您下线的消息。

关键技术

AgileCurse.dll 动态链接库

AgileCurse.dll 是我用C++ 编写的动态链接库，并用python-ctypes进行封装而成的辅助库，主要用来设置控制台的光标显示，光标位置，控制前景色背景色，修改窗口名称，设置窗口大小，改变窗口位置等功能。本次并未完全使用其中的功能。

利用AgileCurse.dll ，程序实现了作者信息在窗口下方滚动播放，以及进度条的定向输出。

搜索局域网ip

这里使用了多线程和子进程管道技术，并行对局域网下的ip进行ping操作，并返回结果。

收发消息

收发消息采用双线程阻塞运行的方法，收消息和发消息分别为两个线程，而收消息会在input输入阻塞时缓存起来，在发消息完成后输出到控制台并释放缓存。当对方或自己结束连接时，使用了互斥锁和信号来触发结束两个线程，也就是说无论收消息线程和发消息线程哪个检验到连接中断都会触发另外一个线程结束。

广域网服务器部署

为了能使不同局域网下的用户能够连接通信，需要中间的代理，我使用华为云的云耀服务器，将服务程序上传到服务器后，被动接收其他用户的连接，并将不同用户发送的消息广播给其他用户，这里为了能够同时连接多个用户，采用多线程技术，同时由于收到服务器内存和性能限制，这里限制最多能连接10个用户。

void  setCursePosition(int x, int y) {
	HANDLE output = GetStdHandle(STD_OUTPUT_HANDLE);
	COORD pos = { x,y };
	SetConsoleCursorPosition(output, pos);
}

void curse::set_character(int row, int column, char ch, bool flush = false) {
	/*---- 在特定位置刷新符号 -----*/
	if (flush) system("cls");
	setCursePosition(row, column);
	printf("%c", (const char*)ch);
}

void curse::flush_with(int from_row, int to_row, int from_column, int to_column, char ch) {
	/*----- 用特定符号刷新特定区域 ------
	
	*/
	for (int i = from_row; i < to_row; i++) {
		for (int j = from_column; j < to_column; j++) {
			setCursePosition(i, j);
			printf("%c", (const char*)ch);
		}
	}
}

void curse::set_title(char* title) {
	SetConsoleTitle((LPCSTR)title);
}


void curse::set_color(int foreground, int background) {
	/*------设置颜色-----
	 0 黑色		A 淡绿
	 1 蓝色     B 浅绿
	 2 绿色     C 淡红
	 3 湖蓝     D 淡紫
	 4 红色		E 淡黄
	 5 紫色		F 亮白
	 6 黄色
	 7 白色
	 8 灰色
	*/
	SetConsoleTextAttribute(GetStdHandle(STD_OUTPUT_HANDLE), (background << 4) + foreground);
}

void curse::set_windowSize(int width, int height) {
	/*------设置窗口大小-------
		宽和高
	*/
	SMALL_RECT wrt = { 0,0,width - 1,height - 1 };
	SetConsoleWindowInfo(GetStdHandle(STD_OUTPUT_HANDLE), TRUE, &wrt);
	COORD coord = { width,height };
	SetConsoleScreenBufferSize(GetStdHandle(STD_OUTPUT_HANDLE), coord);
}

RECT curse::get_windowPos() {
	/*-----获得窗口位置------

	char* title[300];
	GetConsoleTitle((LPSTR)title,(DWORD)299);

	RECT rect;
	HWND hwnd = FindWindow(NULL, (LPCSTR)title);
	GetWindowRect(hwnd, &rect);
	return rect;
}

def find_local_ip():
    myname = getfqdn(gethostname())
    myaddr = gethostbyname(myname)
    return myaddr
        
def find_ip(ip_prefix):
    ip_info = open("ip_info.txt",'w',encoding='utf-8')
    threads = []
    for i in range(1, 256):
        ip = '%s.%s'%(ip_prefix,i)
        threads.append(threading.Thread(target=ping_ip, args=(ip, ip_info)))
    line_clock = threading.Thread(target=progress,args=(255,))
    for i in threads:
        i.start()
    line_clock.start()
    for i in threads:
        i.join()
    line_clock.join()
    print("完成")
    ip_info.close()
    
def progress(final = 30):
    global process_counter
    line = eprogress.LineProgress(total=100,width=30,title="初始化")
    while (process_counter < final):
        time.sleep(0.3)
        screen_lock.acquire()
        set_cursePos(0,29)
        line.update((process_counter*100)//final)
        screen_lock.release()
    screen_lock.acquire()
    line.update(100)
    screen_lock.release()
    process_counter=0
    
def ping_ip(ip_str,ip_info):
    global process_counter,live_ip,ip_info_lock
    cmd = "ping -n 1 %s "%ip_str
    pipe = subprocess.Popen(cmd,shell=True,stdout =subprocess.PIPE).stdout
    output = pipe.readlines()
    for line in output:
        if str(line).upper().find("TTL") >= 0:
        #print("ip: %s 在线" % ip_str)
            ip_info_lock.acquire()
            ip_info.write("ip: %s 在线\n"%ip_str)
            ip_info_lock.release()
            live_ip += 1
            break 
    process_counter += 1
        

def recv_mes(client):
    global recv_box,box_lock,connection,connection_lock
    while(1):
        try:
            content = client.recv(1024)
        except:
            print("连接已经断开")
            connection_lock.acquire()
            connection = False
            connection_lock.release()
            return 
        box_lock.acquire()
        recv_box.append((time.ctime(),content))
        box_lock.release()
        
def send_mes(client):
    global recv_box,box_lock,connection,connection_lock
    connection_lock.acquire()

			print("%s 取消连接"%str(addr))
			log("\ninfo: %addr cancel connection"%str(addr))
			socket_lock.acquire()
			try:
				socket_list.remove(s)
				socket_ip_list.remove(addr[0])
			except:
				pass
			socket_lock.release()
			return None

def socket_target(s,addr):
	# 广播函数
	global socket_list,socket_lock,socket_ip_list
	print("%s 连接"%str(addr))
	log("\ninfo: %addr build connection"%str(addr))
	#try:
	if 1:
		while(1):
			content = read_client(s,addr)
			print(content)
			log(str(content)+ "\tfrom\t"+str(addr))
			info_lock.acquire()
			info_box.append((s,addr,content))
			info_lock.release()
			if content is None:
				break
			else:
				socket_lock.acquire()
				temp = []
				if check(content):
					# 检测失败的内容不广播
					for k,i in enumerate(socket_list):
						if i != s:
							try:
								i.send(content)
							except:
								temp.append([i,socket_ip_list[k]])
				for i,j in temp:
					socket_list.remove(i)
					socket_ip_list.remove(j)
				socket_lock.release()
				time.sleep(0.5)		

def robot():
	# 机器人，用于在无人时回复简单消息，并对检测的ip进行封禁
	global socket_list,socket_lock,info_box,info_lock,socket_ip_list
	history = None
	while(1):
		if history not in socket_list:
			history = None
		socket_lock.acquire()
		if len(socket_list) == 1:
			if socket_list[0] != history:
				# 没人在这！
				socket_list[0].send("robot says: nobody here!".encode())
				history = socket_list[0]
		socket_lock.release()
		info_lock.acquire()
		for i in info_box:
			# 对检测成功的ip连接进行关闭
			if not check(i[2]):
				ip , _ = i[1]
				socket_lock.acquire()
				for k in range(len(socket_list)):
					if ip == socket_ip_list[k]:
						socket_list[k].close()
				socket_lock.release()
				with open("ip.txt",'a',encoding='utf-8') as f:
					# 记录Ip封禁日志
					f.write (str(i[1]).split(',')[0][2:-1]+'\n')
		info_box.clear()
		info_lock.release()
		with open("ip_state.txt",'w',encoding='utf-8') as f:
			for i in socket_ip_list:
				f.write(str(i)+'\n')
		time.sleep(1)
